95  string NormalizeResponse (string jsonResponse)
96  {
97  string nr;
98  if (jsonResponse == String.Empty) return String.Empty;
99 
100  // Translate Json to plain text.
101  nr = jsonResponse.Replace ("\\u003c", "<")
102  .Replace ("\\u003d", "=").Replace ("\\u003e", ">")
103  .Replace ("\\u0026amp;", "&").Replace (@"\""", @"""");
104  try
105  {
106  // We remove the numPage tokens before to compare.
107  nr = nr.Substring (nr.IndexOf (',', nr.IndexOf (',') + 1));
108  nr = nr.Remove(nr.LastIndexOf(','));
109 
110  // Also format empty responses
111  if (nr == ",\"\"") nr = string.Empty;
112 
113  return nr;
114  }
115  catch(Exception e)
116  {
117  // Log and report unknow protocol error exceptions.
118  logger.Warn (Constants.NORMALIZE_REVIEW_RESPONSE_ERROR + e);
119  ExceptionCallback.SendError (
120  Constants.NORMALIZE_REVIEW_RESPONSE_ERROR + e,
121  ref Globals.NORMALIZE_REVIEW_RESPONSE_ERROR_FLAG);
122  return "";
123  }

137  string GetHtmlCode(string appId, int nReviewPage, string lang)
138  {
139  do
140  {
141  #region REGION: REQUEST
142  // Perform a search
143  Thread.Sleep(Constants.WAIT_FOR_REVIEWS);
144  reqUrl = Constants.REVIEWS_URL;
145  reqData = String.Format (Constants.REVIEWS_POST_DATA,
146  nReviewPage, appId, lang);
147  response = NormalizeResponse(server.Post (reqUrl, reqData));
148  Globals.N_REQUEST_ERROR++;
149  totalRequests++;
150  nCurrentPage++;
151 
152  // Set flags
153  requestSucceed = !(String.IsNullOrEmpty (response) &&
154  server.StatusCode == HttpStatusCode.OK);
155  weAreBanned = server.StatusCode == HttpStatusCode.ServiceUnavailable;
156  tooManyConsecutiveBadRequests =
157  nRequestAttemps >= Constants.RETRY_LIMIT;
158  tooManyConsecutiveErrors =
159  Globals.N_REQUEST_ERROR > Constants.MAX_REQUEST_ERRORS;
160  isDonePaging = (response == prevResponse ||
161  nCurrentPage >= Constants.REVIEWS_MAX_PAGE);
162  prevResponse = response;
163  #endregion
164 
165  #region REGION: EXCEPTION CONTROL
166  // Too many consecutive errors of any kind?
167  if (tooManyConsecutiveErrors) { // -> SEND EMAIL
168  logger.Warn (
169  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR);
170  ExceptionCallback.SendError (
171  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR,
172  ref Globals.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR_FLAG);
173  }
174 
175  // Are we banned?
176  if (weAreBanned) {
177  /*Please read:
178  This webservice return an empty response when there's no
179  reviews to return. Be aware.*/
180  logger.Info("BANNED REQUESTING "+appId+": waiting " +
181  Constants.WAIT_ON_BAN + "ms");
182  Thread.Sleep (Constants.WAIT_ON_BAN);
183  }
184 
185  // Request error?
186  if (!requestSucceed) // -> LOG
187  {
188  logger.Warn("REQUEST ERROR: " +
189  server.StatusCode);
190 
191  // Too many consecutived requests failed? -> SKIP.
192  if (tooManyConsecutiveBadRequests)
193  {
194  logger.Warn("TOO MANY CONSECUTIVE "+
195  "BAD REQUESTS. SKIPPING.");
196  break;
197  }
198  }
199  #endregion
200  } while(weAreBanned || !requestSucceed);
201 
202  // Request succeeded? Reset counters
203  Globals.N_REQUEST_ERROR = 0;
204  nRequestAttemps = 0;
205 
206  return response;
207  }

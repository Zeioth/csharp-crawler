756  this.renameDownloadFile = function(filename) {
757  var matches = filename.match(/\s\(\d+\)/g); // find any matches for example (1), (4), (123)
758  if (!matches) { return filename + " (1)"; }
759  var lastMatch = matches[matches.length - 1]; // get the last match, which is the relevant one
760  // make sure it's at the end of the string
761  if (filename.lastIndexOf(lastMatch) !== filename.length-lastMatch.length) {
762  return filename + " (1)";
763  }
764  var fileNum = lastMatch.trim();
765  fileNum = fileNum.substr(1);
766  fileNum = parseInt(fileNum);
767  fileNum++;
768  filename = filename.substring(0, filename.length-lastMatch.length);
769  return filename + " (" + fileNum + ")";

441 function getHandleTypeMatchingFunctionIn(functions, handle) {
442  let {type, subType, className} = getTypes(handle);
443  if (functions["type" + type]) {
444  return functions["type" + type];
445  }
446  if (handle.type === "object" && functions["objectSubType" + subType]) {
447  return functions["objectSubType" + subType];
448  }
449  if (functions["className" + className]) {
450  return functions["className" + className];
451  }
452  if (handle.type === "object") {
453  return functions.default;
454  }
455  return null;

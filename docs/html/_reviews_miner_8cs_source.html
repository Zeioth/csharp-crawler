117  string NormalizeResponse (string jsonResponse)
118  {
119  string nr;
120  if (jsonResponse == String.Empty) return String.Empty;
121 
122  // Translate Json to plain text.
123  nr = jsonResponse.Replace ("\\u003c", "<")
124  .Replace ("\\u003d", "=").Replace ("\\u003e", ">")
125  .Replace ("\\u0026amp;", "&").Replace (@"\""", @"""");
126  try
127  {
128  // We remove the numPage tokens before to compare.
129  nr = nr.Substring (nr.IndexOf (',', nr.IndexOf (',') + 1));
130  nr = nr.Remove(nr.LastIndexOf(','));
131 
132  // Also format empty responses
133  if (nr == ",\"\"") nr = string.Empty;
134 
135  return nr;
136  }
137  catch(Exception e)
138  {
139  // Log and report unknow protocol error exceptions.
140  logger.Warn (Constants.NORMALIZE_REVIEW_RESPONSE_ERROR + e);
141  ExceptionCallback.SendError (
142  Constants.NORMALIZE_REVIEW_RESPONSE_ERROR + e,
143  ref Globals.NORMALIZE_REVIEW_RESPONSE_ERROR_FLAG);
144  return "";

159  string GetHtmlCode(string appId, int nReviewPage, string lang)
160  {
161  do
162  {
163  #region REGION: REQUEST
164  // Perform a search
165  Thread.Sleep(Constants.WAIT_FOR_REVIEWS);
166  reqUrl = Constants.REVIEWS_URL;
167  reqData = String.Format (Constants.REVIEWS_POST_DATA,
168  nReviewPage, appId, lang);
169  response = NormalizeResponse(server.Post (reqUrl, reqData));
170  Globals.N_REQUEST_ERROR++;
171  nRequests++;
172  nCurrentPage++;
173 
174  // Set flags
175  requestSucceed = !(String.IsNullOrEmpty (response) &&
176  server.StatusCode == HttpStatusCode.OK);
177  weAreBanned = server.StatusCode == HttpStatusCode.ServiceUnavailable;
178  tooManyConsecutiveBadRequests =
179  nRequestAttemps >= Constants.RETRY_LIMIT;
180  tooManyConsecutiveErrors =
181  Globals.N_REQUEST_ERROR > Constants.MAX_REQUEST_ERRORS;
182  isDonePaging = (response == prevResponse ||
183  nCurrentPage >= Constants.REVIEWS_MAX_PAGE);
184  prevResponse = response;
185  #endregion
186 
187  #region REGION: EXCEPTION CONTROL
188  // Too many consecutive errors of any kind?
189  if (tooManyConsecutiveErrors) { // -> SEND EMAIL
190  logger.Warn (
191  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR);
192  ExceptionCallback.SendError (
193  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR,
194  ref Globals.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR_FLAG);
195  }
196 
197  // Are we banned?
198  if (weAreBanned) {
199  /*Please read:
200  This webservice return an empty response when there's no
201  reviews to return. Be aware.*/
202  logger.Info("BANNED REQUESTING "+appId+": waiting " +
203  Constants.WAIT_ON_BAN + "ms");
204  Thread.Sleep (Constants.WAIT_ON_BAN);
205  }
206 
207  // Request error?
208  if (!requestSucceed) // -> LOG
209  {
210  logger.Warn("REQUEST ERROR: " +
211  server.StatusCode);
212 
213  // Too many consecutived requests failed? -> SKIP.
214  if (tooManyConsecutiveBadRequests)
215  {
216  logger.Warn("TOO MANY CONSECUTIVE "+
217  "BAD REQUESTS. SKIPPING.");
218  break;
219  }
220  }
221  #endregion
222 
223  } while(weAreBanned || !requestSucceed);
224 
225  // Request succeeded? Reset counters
226  Globals.N_REQUEST_ERROR = 0;
227  nRequestAttemps = 0;
228 
229  return response;

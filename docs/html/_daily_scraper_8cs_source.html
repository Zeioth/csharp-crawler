106  void RunScraper (DailyScraper ds)
107  {
108  // Configure client
109  logger.Info ("Worker Started: Configuring...");
110  if (arg_hpMode) mongoDB.PopulateHPDailyScraperQueue();
111  else mongoDB.PopulateDailyScraperQueue();
112  logger.Info ("PARSING STARTED.");
113 
114  try
115  {
116  while (true)
117  {
118  // Process until queue it empty.
119  if (arg_hpMode) appId = mongoDB.GetFromQueue(
120  Constants.HP_DAILY_SCRAPER_QUEUE_COLLECTION);
121  else appId = mongoDB.GetFromQueue(
122  Constants.DAILY_SCRAPER_QUEUE_COLLECTION);
123  if (appId == null) break;
124 
125  // Get HTML Code
126  response = GetHtmlCode(appId);
127  if (string.IsNullOrEmpty(response)) continue;
128 
129  // Parse it
130  parsedApp = parser.ParseAppPage (response, appId);
131  if (IsEntity(parsedApp)) continue;
132  wBuffer.Add(parsedApp);
133 
134  // Parse related urls
135  bufferRelated.AddRange(
136  parser.ParseRelatedApps (response));
137 
138  // Parse reviews
139  if (arg_parseReviews || arg_parseUsers)
140  authorUrls = ds.reviewsScraper.RunReviewsScraper(
141  parsedApp, usersScraper.ParsedUsers);
142 
143  // Parse users data
144  if (arg_parseUsers && authorUrls.Count > 0)
145  usersScraper.GetUsersData(authorUrls);
146 
147  //Feedback
148  logger.Debug ("");
149  logger.Debug ("APP BUFFERED: " + appId);
150  if (Constants.SCRAPER_DEBUG_B_ENABLED)
151  logger.Debug ("RELATED BUFFERED: " +
152  bufferRelated.Count);
153 
154  // Is buffer full?
155  bufferIsFull = (
156  wBuffer.Count >= Constants.BUFFER_SIZE);
157  relatedBufferIsFull = (
158  bufferRelated.Count >= Constants.BUFFER_SIZE);
159 
160 
161  // If so, write and clean.
162  if (bufferIsFull) StoreIntoDB(wBuffer);
163  if (relatedBufferIsFull) StoreIntoDB(bufferRelated);
164  }
165  // Ends gracefully and buffer is not empty? -> Store.
166  if (wBuffer.Count > 0) StoreIntoDB(wBuffer);
167  if (bufferRelated.Count > 0) StoreIntoDB(bufferRelated);
168  }
169  catch(Exception e)
170  {
171  logger.Error(Constants.SCRAPER_ERROR + e);
172  ExceptionCallback.SendError(Constants.SCRAPER_ERROR + e);
173  }
174  finally
175  {
176  // Close firefox
177  if (firefoxDriver != null) firefoxDriver.Quit();
178 
179  // Good bye!
180  logger.Info ("GOOGLE PLAY DAILY SCRAPER FINISHED: "
181  + nBlocksProcessed + " Blocks of " + Constants.BUFFER_SIZE +
182  " apps added to apps data.");
183  }
184  }

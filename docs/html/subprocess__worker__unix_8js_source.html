142 function writePipe(pipe, data) {
143 
144  postMessage({msg: "debug", data: "trying to write to "+pipe});
145 
146  let numChunks = Math.floor(data.length / BufferSize);
147  let pData = new WriteBuffer();
148 
149  for (var chunk = 0; chunk <= numChunks; chunk ++) {
150  let numBytes = chunk < numChunks ? BufferSize : data.length - chunk * BufferSize;
151  for (var i=0; i < numBytes; i++) {
152  pData[i] = data.charCodeAt(chunk * BufferSize + i) % 256;
153  }
154 
155  let bytesWritten = libcFunc.write(pipe, pData, numBytes);
156  if (bytesWritten != numBytes) {
157  closePipe();
158  libc.close();
159  postMessage({ msg: "error", data: "error: wrote "+bytesWritten+" instead of "+numBytes+" bytes"});
160  close();
161  }
162  }
163  postMessage({msg: "info", data: "wrote "+data.length+" bytes of data"});
164 }

179 function readPipe(pipe, charset, pid) {
180  var p = new libcFunc.pollFds;
181  p[0].fd = pipe;
182  p[0].events = POLLIN | POLLERR | POLLHUP;
183  p[0].revents = 0;
184  var pollTimeout = WAITTIME;
185  var exitCode = -1;
186  var readCount = 0;
187  var result, status = ctypes.int();
188  result = 0;
189 
190 
191  const i=0;
192  while (true) {
193  if (result == 0) {
194  result = libcFunc.waitpid(pid, status.address(), WNOHANG);
195  if (result > 0) {
196  pollTimeout = NOWAIT;
197  exitCode = parseInt(status.value);
198  postMessage({msg: "debug", data: "waitpid signaled subprocess stop, exitcode="+status.value });
199  }
200  }
201  var r = libcFunc.poll(p, 1, pollTimeout);
202  if (r > 0) {
203  if (p[i].revents & POLLIN) {
204  postMessage({msg: "debug", data: "reading next chunk"});
205  readCount = readPolledFd(p[i].fd, charset);
206  if (readCount == 0) break;
207  }
208 
209  if (p[i].revents & POLLHUP) {
210  postMessage({msg: "debug", data: "poll returned HUP"});
211  break;
212  }
213  else if (p[i].revents & POLLERR) {
214  postMessage({msg: "error", data: "poll returned error"});
215  break;
216  }
217  else if (p[i].revents != POLLIN) {
218  postMessage({msg: "error", data: "poll returned "+p[i]});
219  break;
220  }
221  }
222  else
223  if (pollTimeout == 0 || r < 0) break;
224  }
225 
226  // continue reading until the buffer is empty
227  while (readCount > 0) {
228  readCount = readPolledFd(pipe, charset);
229  }
230 
231  libcFunc.close(pipe);
232  postMessage({msg: "done", data: exitCode });
233  libc.close();
234  close();
235 }

237 function readPolledFd(pipe, charset) {
238  var line = new ReadBuffer();
239  var r = libcFunc.read(pipe, line, BufferSize);
240 
241  if (r > 0) {
242  var c = readString(line, r, charset);
243  postMessage({msg: "data", data: c, count: c.length});
244  }

248 onmessage = function (event) {
249  switch (event.data.msg) {
250  case "init":
251  initLibc(event.data.libc);
252  break;
253  case "read":
254  initLibc(event.data.libc);
255  readPipe(event.data.pipe, event.data.charset, event.data.pid);
256  break;
257  case "write":
258  // data contents:
259  // msg: 'write'
260  // data: the data (string) to write
261  // pipe: ptr to pipe
262  writePipe(event.data.pipe, event.data.data);
263  postMessage({msg: "info", data: "WriteOK"});
264  break;
265  case "close":
266  postMessage({msg: "debug", data: "closing stdin\n"});
267

59  string GetHtmlCode(string rankingUrl)
60  {
61  do
62  {
63  #region REGION: REQUEST
64  // Perform a search
65  Thread.Sleep (Constants.WAIT_FOR_TOP);
66  reqUrl = String.Format (rankingUrl);
67  reqData = String.Format (
68  Constants.CRAWL_CAT_POST_DATA,
69  (nCurrentPage*Constants.TOP_CRAWLER_PAGE_RESULTS),
70  Constants.TOP_CRAWLER_PAGE_RESULTS
71  );
72  response = server.Post (reqUrl, reqData);
73  Globals.N_REQUEST_ERROR++;
74  totalRequests++;
75  nCurrentPage++;
76 
77  // Set flags
78  requestSucceed = !(String.IsNullOrEmpty (response) &&
79  server.StatusCode == HttpStatusCode.OK);
80  weAreBanned = server.StatusCode == HttpStatusCode.ServiceUnavailable;
81  tooManyConsecutiveBadRequests =
82  nRequestAttemps >= Constants.RETRY_LIMIT;
83  tooManyConsecutiveErrors =
84  Globals.N_REQUEST_ERROR > Constants.MAX_REQUEST_ERRORS;
85  isDonePaging = nCurrentPage >= Constants.TOP_CRAWLER_MAX_PAGES;
86  #endregion
87 
88  #region REGION: EXCEPTION CONTROL
89  // Too many consecutive errors of any kind?
90  if (tooManyConsecutiveErrors) { // -> SEND EMAIL
91  logger.Warn (
92  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR);
93  ExceptionCallback.SendError (
94  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR,
95  ref Globals.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR_FLAG);
96  }
97 
98  // Are we banned?
99  if (weAreBanned) { // -> WAIT
100  logger.Warn ("BANNED: after " + totalRequests +
101  " requests -> waiting " + Constants.WAIT_ON_BAN + "ms");
102  Thread.Sleep (Constants.WAIT_ON_BAN);
103  }
104 
105  // Request error?
106  if (!requestSucceed) // -> LOG
107  {
108  logger.Warn("REQUEST ERROR: " +
109  server.StatusCode);
110 
111  // Too many consecutived requests failed? -> SKIP.
112  if (tooManyConsecutiveBadRequests)
113  {
114  logger.Warn("TOO MANY CONSECUTIVE "+
115  "BAD REQUESTS. SKIPPING.");
116  break;
117  }
118  }
119  #endregion
120 
121  } while(weAreBanned || !requestSucceed);
122 
123  // Request succeeded? Reset counters
124  Globals.N_REQUEST_ERROR = 0;
125  nRequestAttemps = 0;
126 
127  return response;
128  }

137  void CrawlRanking (string country, string rankUrl,
138  string rankName, string field)
139  {
140  logger.Debug ("Crawling Ranking : [ " + rankName + " ] "+
141  "for " + country);
142  do
143  {
144  // Get HTML Code
145  response = GetHtmlCode (rankUrl);

147 
148  // Parse it
149  parsedUrls = parser.ParseSearchResults(response);
150  wBuffer.AddRange(parsedUrls);
151 
152  // Feedback
153  logger.Debug("Current page: " + nCurrentPage);
154  logger.Info("Apps found: " + parsedUrls.Count);
155 
156  // Is buffer full? -> Write and clean.
157  bufferIsFull = (wBuffer.Count >= Constants.BUFFER_SIZE);
158  if (bufferIsFull) StoreIntoDB(wBuffer, country, field);
159 
160  } while(!isDonePaging);
161 
162  // Ends gracefully and buffer is not empty? -> Store.
163  if (wBuffer.Count > 0) StoreIntoDB(wBuffer, country, field);
164 
165  // Reset object
166  ResetObject ();

478  pull: function adb_pull(aFrom, aDest) {

484  let remaining;

486  let chunkSize;

503  // extract chunk data header info. to headerArray.
504  let extractChunkDataHeader = function(data) {
505  let tmpArray = new Uint8Array(headerArray.buffer);
506  for (let i = 0; i < 8 - currentHeaderLength; i++) {
507  tmpArray[currentHeaderLength + i] = data[i];
508  }
509  }
510 
511  // chunk data header is 8 bytes length,
512  // the first 4 bytes: hex4("DATA"), and
513  // the second 4 bytes: hex4(chunk size)
514  let checkChunkDataHeader = function(data) {
515  if (data.length + currentHeaderLength >= 8) {
516  extractChunkDataHeader(data);
517 
518  if (headerArray[0] != DATA) {
519  shutdown();
520  return false;
521  }
522  // remove header info. from socket package data
523  pkgData = data.subarray(8 - currentHeaderLength, data.length);
524  chunkSize = headerArray[1];
525  currentHeaderLength = 0;
526  return true;
527  }
528 
529  // If chunk data header info. is separated into more than one
530  // socket package, keep partial header info. in headerArray.
531  let tmpArray = new Uint8Array(headerArray.buffer);
532  for (let i = 0; i < data.length; i++) {
533  tmpArray[currentHeaderLength + i] = data[i];
534  }
535  currentHeaderLength += data.length;
536  return true;
537  }
538 
539  // The last remaining package data contains 8 bytes,
540  // they are "DONE(0x454e4f44)" and 0x0000.
541  let checkDone = function(data) {
542  if (remaining != 0 || data.length != 8) {
543  return false;
544  }
545 
546  let doneFlagArray = new Uint32Array(1);
547  let tmpArray = new Uint8Array(doneFlagArray.buffer);
548  for (let i = 0; i < 4; i++) {
549  tmpArray[i] = data[i];
550  }
551  // Check DONE flag
552  if (doneFlagArray[0] == DONE) {
553  return true;
554  }
555  return false;

558  let runFSM = function runFSM(aData) {
559  console.log("runFSM " + state);
560  let req;
561  switch(state) {
562  case "start":
563  state = "send-transport";
564  runFSM();
565  break;
566  case "send-transport":
567  req = ADB._createRequest("host:transport-any");
568  ADB.sockSend(socket, req);
569  state = "wait-transport";
570  break;
571  case "wait-transport":
572  if (!ADB._checkResponse(aData, OKAY)) {
573  shutdown();
574  return;
575  }
576  console.log("transport: OK");
577  state = "send-sync";
578  runFSM();
579  break;
580  case "send-sync":
581  req = ADB._createRequest("sync:");
582  ADB.sockSend(socket, req);
583  state = "wait-sync";
584  break;
585  case "wait-sync":
586  if (!ADB._checkResponse(aData, OKAY)) {
587  shutdown();
588  return;
589  }
590  console.log("sync: OK");
591  state = "send-stat";
592  runFSM();
593  break;
594  case "send-stat":
595  infoLengthPacket = new Uint32Array(1);
596  infoLengthPacket[0] = aFrom.length;
597  ADB.sockSend(socket, encoder.encode("STAT"));
598  ADB.sockSend(socket, infoLengthPacket);
599  ADB.sockSend(socket, encoder.encode(aFrom));
600 
601  state = "wait-stat";
602  break;
603  case "wait-stat":
604  if (!ADB._checkResponse(aData, STAT)) {
605  shutdown();
606  return;
607  }
608  view = new Uint32Array(aData);
609  // view contains 4 elements of 32 bit;
610  // the first element is hex "STAT"
611  // the second one is file mode
612  // the third one is file length
613  // and the last one is file time (lastModified?)
614  fileSize = view[2];
615  chunkSize = 0;
616  remaining = fileSize;
617  fileData = new Uint8Array(new ArrayBuffer(fileSize));
618  headerArray = new Uint32Array(2);
619  currentPos = 0;
620  currentHeaderLength = 0;
621  console.log("stat: OK");
622  state = "send-recv";
623  runFSM();
624  break;
625  case "send-recv":
626  infoLengthPacket = new Uint32Array(1);
627  infoLengthPacket[0] = aFrom.length;
628  ADB.sockSend(socket, encoder.encode("RECV"));
629  ADB.sockSend(socket, infoLengthPacket);
630  ADB.sockSend(socket, encoder.encode(aFrom));
631 
632  state = "wait-recv";
633  break;
634  case "wait-recv":
635  // After sending "RECV" command, adb server will send chunks data back,
636  // Handle every single socket package here.
637  // Note: One socket package maybe contain many chunks, and often
638  // partial chunk at the end.
639  pkgData = new Uint8Array(aData);
640 
641  // Handle all data in a single socket package.
642  while(pkgData.length > 0) {
643  if (chunkSize == 0 && checkDone(pkgData)) {
644  OS.File.writeAtomic(aDest, fileData, {}).then(
645  function onSuccess(number) {
646  console.log(number);
647  deferred.resolve("SUCCESS");
648  },
649  function onFailure(reason) {
650  console.log(reason);
651  deferred.reject("CANT_ACCESS_FILE");
652  }
653  );
654 
655  state = "send-quit";
656  runFSM();
657  return;
658  }
659  if (chunkSize == 0 && !checkChunkDataHeader(pkgData)) {
660  shutdown();
661  return;
662  }
663  // handle full chunk
664  if (chunkSize > 0 && pkgData.length >= chunkSize) {
665  let chunkData = pkgData.subarray(0, chunkSize);
666  fileData.set(chunkData, currentPos);
667  pkgData = pkgData.subarray(chunkSize, pkgData.length);
668  currentPos += chunkSize;
669  remaining -= chunkSize;
670  chunkSize = 0;
671  }
672  // handle partial chunk at the end of socket package
673  if (chunkSize > 0 && pkgData.length < chunkSize) {
674  fileData.set(pkgData, currentPos);
675  currentPos += pkgData.length;
676  remaining -= pkgData.length;
677  chunkSize -= pkgData.length;
678  break; // Break while loop.
679  }
680  }
681 
682  break;
683  case "send-quit":
684  infoLengthPacket = new Uint32Array(1);
685  infoLengthPacket[0] = 0;
686  ADB.sockSend(socket, encoder.encode("QUIT"));
687  ADB.sockSend(socket, infoLengthPacket);
688 
689  state = "end";
690  runFSM();
691  break;
692  case "end":
693  socket.close();
694  break;
695  default:
696  console.log("pull Unexpected State: " + state);
697  deferred.reject("UNEXPECTED_STATE");
698  }
699  }

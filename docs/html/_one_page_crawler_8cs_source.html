207  private static void CrawlOnePage (MongoDBWrapper mongoDB, ref WebRequestWrapper server,
208  PlayStoreParser parser,
209  string searchField)
210  {
211  #region REGION: SET VARIABLES AND INSTANCES
212  string response; // HTML content
213  #endregion
214 
215  #region REGION: CRAWL THE FIRST PAGE
216  // REQUESTS FOR CRAWLSTORE
217  logger.Debug ("Crawling Search Term : [ " + searchField + " ]");
218  do
219  {
220  response = server.Post (String.Format (Constants.CRAWL_URL, searchField), Constants.INITIAL_POST_DATA);
221  nRequests++;
222 
223  // BANNED? -> WAIT
224  if (parser.ParseSearchResults(response).Count() <= 0)
225  {
226  // TOO MANY CONSECUTIVE ERRORS? -> SEND EMAIL
227  nConsecutiveErrors++;
228  if (nConsecutiveErrors >= Constants.MAX_REQUEST_ERRORS)
229  {
230  logger.Warn(Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR);
231  ExceptionCallback.SendError (Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR,
232  ref Globals.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR_FLAG);
233  }
234  logger.Info("BANNED: after " + nRequests + " requests -> waiting " + Constants.WAIT_ON_BAN + "ms");
235  Thread.Sleep(Constants.WAIT_ON_BAN);
236  }
237  } while(parser.ParseSearchResults(response).Count() <= 0);
238 
239  // ENQUEUE
240  EnqueueOne(mongoDB, parser, response);
241  mongoDB.Remove (searchField, Constants.OPC_QUEUE_COLLECTION, "Term");
242 
243  Thread.Sleep (Constants.WAIT);
244  logger.Debug("PAGE CRAWL SUCEED");
245  #endregion
246  }

255  private static void EnqueueOne (MongoDBWrapper mongoDB, PlayStoreParser parser, string response)
256  {
257  foreach (string url in parser.ParseSearchResults (response))
258  {
259  // IS NOT DUPLICATED? -> ENQUEUE IT
260  if ((!mongoDB.DocumentAlreadyExists (Constants.APP_URL_PREFIX + url, Constants.APPS_DATA_COLLECTION))
261  && (!mongoDB.IsAlreadyQueued (url, Constants.SCRAPER_QUEUE_COLLECTION)))
262  {
263  mongoDB.UpsertReplacing (new QueuedApp (url), Constants.SCRAPER_QUEUE_COLLECTION);
264  logger.Debug ("Queued App - " + url);
265  nQueuedApps++;
266  } else
267  logger.Debug ("Duplicated App. Skipped - " + url);
268  }
269  }

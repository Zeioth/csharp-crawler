115  let runtimes = [];

117  let b2gExists = yield device.shell(query);
118  // XXX: Sometimes we get an empty response back. Likely a bug in our shell
119  // code in this add-on.
120  // There are also some Android devices that do not have `test` installed.
121  for (let attempts = 3; attempts > 0; attempts--) {
122  b2gExists = yield device.shell(query);

147 // This requires Unix socket support from Firefox for Android (35+)
148 FirefoxOnAndroidRuntime.detect = Task.async(function*(device, model) {
149  let runtimes = [];
150  // A matching entry looks like:
151  // 00000000: 00000002 00000000 00010000 0001 01 6551588 /data/data/org.mozilla.fennec/firefox-debugger-socket
152  let query = "cat /proc/net/unix";
153  let rawSocketInfo = yield device.shell(query);
154  let socketInfos = rawSocketInfo.split("\r\n");
155  // Filter to lines with "firefox-debugger-socket"
156  socketInfos = socketInfos.filter(l => l.contains("firefox-debugger-socket"));
157  // It's possible to have multiple lines with the same path, so de-dupe them
158  let socketPaths = new Set();
159  for (let socketInfo of socketInfos) {
160  let socketPath = socketInfo.split(" ").pop();
161  socketPaths.add(socketPath);
162  }
163  for (let socketPath of socketPaths) {
164  let runtime = new FirefoxOnAndroidRuntime(device, model, socketPath);
165  console.log("Found " + runtime.name);
166  runtimes.push(runtime);
167  }
168  return runtimes;
169 });

48  string GetHtmlCode(string searchField)

57  if (nCurrentPage < 1) reqData = Constants.CRAWL_INIT_POST_DATA;
58  response = server.Post (reqUrl, reqData);
59  Globals.N_REQUEST_ERROR++;
60  nRequests++;
61  nCurrentPage++;
62 
63  // Set flags
64  requestSucceed = (!String.IsNullOrEmpty (response) &&
65  server.HttpStatusCode == HttpStatusCode.OK &&
66  server.WebExceptionStatus == WebExceptionStatus.Success);
67  weAreBanned = server.HttpStatusCode == HttpStatusCode.ServiceUnavailable ||
68  server.HttpStatusCode == HttpStatusCode.NotFound;
69  tooManyConsecutiveBadRequests =
70  nRequestAttemps >= Constants.RETRY_LIMIT;
71  tooManyConsecutiveErrors =
72  Globals.N_REQUEST_ERROR > Constants.MAX_REQUEST_ERRORS;
73  thereIsSearchResults = !response.Contains (
74  Constants.SEARCH_CHECK_STRING);
75  isDonePaging = (nCurrentPage >= Constants.CRAWLER_MAX_PAGES);
76 
77  // Set reqData for the next request (using next page token)
78  rgxMatch = Constants.CRAWL_PAG_TOKEN.Match (response);
79  if (rgxMatch.Success) {
80  thereIsPageToken = true;
81  pagToken = rgxMatch.Value.Replace (":S:", "%3AS%3A")
82  .Replace ("\\x22", string.Empty);
83  reqData = String.Format (Constants.CRAWL_POST_DATA, pagToken);
84  } else {
85  thereIsPageToken = false;
86  logger.Warn ("WARNING: Couldn't find page token in HTML code.");
87  }
88  #endregion
89 
90  #region REGION: EXCEPTION CONTROL
91  // Too many consecutive errors of any kind?
92  if (tooManyConsecutiveErrors) { // -> SEND EMAIL
93  logger.Warn (
94  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR +
95  "HTTP STATUS: " + server.HttpStatusCode +
96  "WEB EXCEPTION: " + server.WebExceptionStatus);
97  ExceptionCallback.SendError (
98  Constants.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR +
99  "HTTP STATUS: " + server.HttpStatusCode +
100  "WEB EXCEPTION: " + server.WebExceptionStatus,
101  ref Globals.TOO_MANY_CONSECUTIVE_REQUESTS_ERROR_FLAG);
102  }
103 
104  // Theres no search results?
105  if (!thereIsSearchResults) { // -> RETURN EMPTY
106  Globals.N_REQUEST_ERROR++;
107  logger.Warn("WARNING: There's no results for this search term.");
108  return string.Empty;
109  }
110 
111  // Are we banned?
112  if (weAreBanned) { // -> WAIT
113  logger.Warn ("BANNED: after " + nRequests +
114  " requests -> waiting " + Constants.WAIT_ON_BAN + "ms");
115  Thread.Sleep (Constants.WAIT_ON_BAN);
116  }
117 
118  // Request error?
119  if (!requestSucceed) { // -> LOG
120  if (server.HttpStatusCode != HttpStatusCode.OK)
121  logger.Warn("REQUEST ERROR: Http status '" +
122  server.HttpStatusCode+"'");
123  if (server.WebExceptionStatus != WebExceptionStatus.Success)
124  logger.Warn("REQUEST ERROR: Web status '" +
125  server.WebExceptionStatus+"'");
126  }
127  #endregion
128 
129  } while(weAreBanned || !requestSucceed);
130 
131  // Request succeeded? Reset counters
132  Globals.N_REQUEST_ERROR = 0;
133  nRequestAttemps = 0;
134 
135  return response;

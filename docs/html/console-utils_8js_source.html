53  c = aStr[i];
54 
55  switch (state) {
56  // Normal JS state.
57  case STATE_NORMAL:
58  if (c == '"') {
59  state = STATE_DQUOTE;
60  }
61  else if (c == "'") {
62  state = STATE_QUOTE;
63  }
64  else if (c == ";") {
65  start = i + 1;
66  }
67  else if (c == " ") {
68  start = i + 1;
69  }
70  else if (OPEN_BODY.indexOf(c) != -1) {

72  token: c,
73  start: start

83  }

85  start = i + 1;
86  }
87  else {
88  start = last.start;
89  }
90  }
91  break;
92

94  case STATE_DQUOTE:
95  if (c == "\\") {

97  }

99  return {

102  }

104  state = STATE_NORMAL;
105  }

107

271 {

273  let numProps = 0;
274 
275  // We need to go up the prototype chain.
276  while (aObj) {
277  let protoAndProps = yield aObj.prototypeAndProperties();
278  if (!protoAndProps) {
279  break;
280  }

282  numProps += props.length;

284  for (let i = 0; i < props.length; i++) {
285  let prop = props[i];

287  continue;
288  }

290  continue;
291  }
292

294  // This uses a trick: converting a string to a number yields NaN if
295  // the operation failed, and NaN is not equal to itself.

297  matches.add(prop);

302  }
